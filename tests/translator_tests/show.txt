int fibonacci(int n) {

    if (n == 1) {
        return 1;
    }

    if (n == 2) {
        return 1;
    }

    int a = 1;
    int b = 1;

    int switch = 0;

    int counter = 2;

    while (counter != n) {

        if (switch == 0) {
            a = a + b;
            switch = 1;
        }

        if (switch == 1) {
            b = b + a;
            switch = 0;
        }

        n = n + 1;
    }

    if (switch == 0) {
        return b;
    }

    if (switch == 1) {
        return a;
    }
}

int main() {

    int a = fibonacci(10);
    return 0;
}
